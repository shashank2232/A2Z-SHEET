#include <bits/stdc++.h>
using namespace std;

int main()
{
   int n;
   cin>>n;
   for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
        // this means whenever we're standing at boundary we have to print * , else print spaces
        if(i == 0 || i == n-1 || j == 0 || j == n-1){
            cout<<"*";
        }
        else {
            cout<<" ";
        }
    }cout<<endl;
   }
   return 0;
}
